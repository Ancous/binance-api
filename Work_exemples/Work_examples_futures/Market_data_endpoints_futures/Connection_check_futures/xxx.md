class TradingThread(QThread):

        self.server = server  # 'test' –∏–ª–∏ 'live'
        self.strategy = strategy
        self.settings = settings
        self.running = True

    def run(self):
        self.log_signal.emit("–ó–∞–ø—É—Å–∫ —Ç–æ—Ä–≥–æ–≤–ª–∏...")
        # –ü—Ä–∏–º–µ—Ä: –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è API
        if self.exchange == "Bybit":

                'apiKey': self.settings['api_key'],
                'secret': self.settings['secret'],
                'sandbox': self.server == "–¢–µ—Å—Ç–æ–≤—ã–π",  # –¢–µ—Å—Ç–æ–≤—ã–π —Ä–µ–∂–∏–º
            })
        # –°–∏–º—É–ª—è—Ü–∏—è —Ç–æ—Ä–≥–æ–≤–ª–∏
        while self.running:
            self.log_signal.emit(f"–ü—Ä–æ–≤–µ—Ä–∫–∞ —Ä—ã–Ω–∫–∞ –¥–ª—è {self.strategy}...")
            time.sleep(5)  # –ó–∞–¥–µ—Ä–∂–∫–∞
        self.log_signal.emit("–¢–æ—Ä–≥–æ–≤–ª—è –æ—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∞.")

    def stop(self):
        self.running = False
